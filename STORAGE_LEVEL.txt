CREATE TABLE SA_CUSTOMERS.SA_CUSTOMERS (
CUSTOMER_ID NUMBER,
CUSTOMER_DESC CHAR(50),
CUSTOMER_FIRST_NAME CHAR(20),
CUSTOMER_LAST_NAME CHAR(20),
CUSTOMER_AGE NUMBER,
CUSTOMER_GENDER CHAR(20),
CONSTRAINT PK_SA_CUSTOMERS PRIMARY KEY (CUSTOMER_ID) );


CREATE TABLE SA_SALES.SA_SALES (
SALE_ID NUMBER,
SALE_FCT_QUANTITY NUMBER,
SALE_FCT_TOTAL_PRICE NUMBER,
CONSTRAINT PK_SA_SALES PRIMARY KEY (SALE_ID) );


CREATE TABLE SA_EMPLOYEE.SA_EMPLOYEE (
EMPLOYEE_ID NUMBER,
EMPLOYEE_DESC CHAR(50),
EMPLOYEE_FIRST_NAME CHAR(20),
EMPLOYEE_LAST_NAME CHAR(20),
CONSTRAINT PK_SA_EMPLOYEE PRIMARY KEY (EMPLOYEE_ID) );


CREATE TABLE SA_BIKES.SA_BIKES (
BIKE_ID NUMBER,
BRAND_NAME CHAR(20),
MODEL_NAME CHAR(20),
SPEC_NAME CHAR(30),
SPEC_ID NUMBER,
SERIAL_NUM NUMBER,
CONSTRAINT PK_SA_BIKES PRIMARY KEY (BIKE_ID) );
